DECLARE students AS dictionary        // key: student_id, value: {password, name, faculty, program}
DECLARE lecturers AS dictionary       // key: username, value: {password}
DECLARE grades AS list                 // each element is a grade record (dictionary)

// 1. Weighted total + letter grade (A if total > 90)
FUNCTION calculate_total_and_letter(test, assign, project, exam)
    total = test*0.20 + assign*0.15 + project*0.30 + exam*0.35
    total = round(total, 2)
    IF total > 90 THEN letter = "A"
    ELSE IF total >= 80 THEN letter = "B"
    ELSE IF total >= 70 THEN letter = "C"
    ELSE IF total >= 60 THEN letter = "D"
    ELSE letter = "F"
    RETURN total, letter
END FUNCTION

// 2. Add or update a grade record
FUNCTION add_grade(student_id, semester, subject, test, assign, project, exam)
    total, letter = calculate_total_and_letter(test, assign, project, exam)
    FOR EACH grade IN grades
        IF grade.student_id == student_id AND grade.semester == semester AND grade.subject == subject THEN
            REMOVE grade FROM grades
        END IF
    END FOR
    APPEND {student_id, semester, subject, test, assign, project, exam, total, letter} TO grades
END FUNCTION

// 3. Get all grades for one student
FUNCTION get_student_grades(student_id)
    result = []
    FOR EACH grade IN grades
        IF grade.student_id == student_id THEN
            APPEND (grade.semester, grade.subject, grade.test, grade.assign, grade.project, grade.exam, grade.total, grade.letter) TO result
        END IF
    END FOR
    RETURN result
END FUNCTION

// 4. Student login validation
FUNCTION student_login_ok(student_id, password)
    IF students CONTAINS student_id AND students[student_id].password == password THEN
        RETURN True
    ELSE
        RETURN False
    END IF
END FUNCTION

// 5. Lecturer login validation
FUNCTION lecturer_login_ok(username, password)
    IF lecturers CONTAINS username AND lecturers[username].password == password THEN
        RETURN True
    ELSE
        RETURN False
    END IF
END FUNCTION

// 6. Register a new student
FUNCTION register_student(student_id, password, name, faculty, program)
    IF student_id IN students THEN RETURN False
    students[student_id] = {password: password, name: name, faculty: faculty, program: program}
    RETURN True
END FUNCTION

// 7. Register a new lecturer
FUNCTION register_lecturer(username, password)
    IF username IN lecturers THEN RETURN False
    lecturers[username] = {password: password}
    RETURN True
END FUNCTION

// 8. Display lecturer dashboard
PROCEDURE lecturer_dashboard(parent_window, username)

    CREATE new window "win" as child of parent_window
    SET title = "Lecturer Dashboard – " + username
    SET size = 850x550

    DISPLAY "Upload Student Grade" heading

    // --- Input form ---
    CREATE dropdown for Semester (Semester 1 to 6)
    CREATE dropdown for Subject (Structured Programming, Database, ...)
    CREATE text field for Student ID
    CREATE text fields for Test (20%), Assignment (15%), Project (30%), Exam (35%)

    CREATE label to show total and grade preview

    // --- Table showing all grades in system ---
    CREATE table with columns:
        Student ID, Semester, Subject, Test, Assign, Project, Exam, Total, Grade

    DEFINE refresh_all_grades:
        CLEAR all rows from table
        FOR EACH grade IN grades
            INSERT row with grade data
        END FOR
    END PROCEDURE

    // --- Preview button logic ---
    DEFINE preview:
        TRY to read test, assign, project, exam as numbers
        IF any number is invalid or not between 0-100 THEN
            SHOW error message
        ELSE
            total, letter = calculate_total_and_letter(test, assign, project, exam)
            UPDATE preview label to show total and letter
        END IF
    END PROCEDURE

    // --- Save button logic ---
    DEFINE save:
        READ student_id from field
        IF student_id NOT in students THEN
            SHOW error "Student not found"
            RETURN
        END IF

        TRY to read test, assign, project, exam as numbers
        IF any number invalid or outside 0-100 THEN
            SHOW error "Enter numbers 0-100"
            RETURN
        END IF

        CALL add_grade(student_id, semester, subject, test, assign, project, exam)
        SHOW success message
        CALL refresh_all_grades()
        CLEAR all input fields
        RESET preview label
    END PROCEDURE

    CREATE buttons:
        - "Preview Grade" → calls preview
        - "Save Grade" → calls save
        - "Refresh List" → calls refresh_all_grades
        - "Logout" → closes win

    CALL refresh_all_grades()

    WAIT for user interaction

END PROCEDURE

// 9. Display Student dashboard
PROCEDURE student_dashboard(parent_window, student_id)

    CREATE new window "win" as child of parent_window
    SET title = "Student Dashboard – " + student_id
    SET size = 750x400

    GET student record from students using student_id
    DISPLAY welcome message with student.name and student.program

    CREATE a table (treeview) with columns:
        Semester, Subject, Test(20%), Assign(15%), Project(30%), Exam(35%), Total, Grade

    DEFINE refresh procedure:
        CLEAR all rows from table
        CALL get_student_grades(student_id) -> store in grade_list
        IF grade_list is empty THEN
            INSERT one row showing "No grades uploaded yet"
        ELSE
            FOR EACH grade IN grade_list
                INSERT row with grade data
            END FOR
        END IF
    END PROCEDURE

    CALL refresh()

    CREATE buttons:
        - "Refresh" button → calls refresh()
        - "Logout" button → closes win

    WAIT for user interaction (window loop)

END PROCEDURE

PROCEDURE main()

    CREATE root window
    SET title = "Academic Portal - Sierra Leone (SDG4)"
    SET size = 500x500

    // --- Create tabs ---
    CREATE notebook (tab control)

    // ---------- LOGIN TAB ----------
    CREATE login_tab inside notebook

    CREATE dropdown for Role (Student / Lecturer)
    CREATE text field for ID/Username
    CREATE password field (masked)

    DEFINE do_login:
        READ role, uid, password
        IF any field empty THEN
            SHOW error "Fill both fields"
            RETURN
        END IF

        IF role == "Student" THEN
            IF student_login_ok(uid, password) THEN
                CALL student_dashboard(root, uid)
                CLEAR login fields
            ELSE
                SHOW error "Invalid Student ID or Password"
            END IF
        ELSE  // Lecturer
            IF lecturer_login_ok(uid, password) THEN
                CALL lecturer_dashboard(root, uid)
                CLEAR login fields
            ELSE
                SHOW error "Invalid Lecturer credentials"
            END IF
        END IF
    END PROCEDURE

    CREATE button "Login" → calls do_login

    // ---------- STUDENT SIGNUP TAB ----------
    CREATE student_tab inside notebook

    CREATE text fields: Student ID, Full Name, Password, Confirm Password
    CREATE readonly field for Faculty (default "Faculty of ICT")
    CREATE dropdown for Program (BSEM, BIT, BBIT, DIT, CIT)

    DEFINE do_student_signup:
        READ student_id, name, password, confirm, faculty, program
        IF student_id, name, or password empty THEN
            SHOW error "ID, Name and Password required"
            RETURN
        END IF
        IF password != confirm THEN
            SHOW error "Passwords do not match"
            RETURN
        END IF
        IF register_student(student_id, password, name, faculty, program) THEN
            SHOW success message
            CLEAR student signup fields
        ELSE
            SHOW error "Student ID already exists"
        END IF
    END PROCEDURE

    CREATE button "Register Student" → calls do_student_signup

    // ---------- LECTURER SIGNUP TAB ----------
    CREATE lecturer_tab inside notebook

    CREATE text fields: Username, Password, Confirm Password

    DEFINE do_lecturer_signup:
        READ username, password, confirm
        IF username or password empty THEN
            SHOW error "Username and password required"
            RETURN
        END IF
        IF password != confirm THEN
            SHOW error "Passwords do not match"
            RETURN
        END IF
        IF register_lecturer(username, password) THEN
            SHOW success message
            CLEAR lecturer signup fields
        ELSE
            SHOW error "Username already taken"
        END IF
    END PROCEDURE

    CREATE button "Register Lecturer" → calls do_lecturer_signup

    // --- Start the GUI event loop ---
    START mainloop of root window

END PROCEDURE
